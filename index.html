# the boring code that earns its keep
import duckdb, polars as pl

def load_daily(src: str) -> pl.DataFrame:
    con = duckdb.connect(":memory:")
    return con.sql(f"""
        SELECT date_trunc('day', ts) AS d,
               count(*) AS events
        FROM read_parquet('{src}')
        GROUP BY 1 ORDER BY 1
    """).pl()   # 40-node Spark job → one laptop
